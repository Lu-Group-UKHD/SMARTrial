Preprocessing data

Subset samples:
Only use AML with chemotherapy without CD34 enrichment, PD and SD will be considered as non-responders (NR) and will be compared to responders (R) This is different to the SMARTrial study, in which only PD were compared to responders. If we only use R here, there are not enough samples.

#dataset
load("../output/protSmart_new.RData")

protSub <- protSmart_raw[, protSmart_raw$DISEASE %in% "AML" & protSmart_raw$chemo_pat %in% TRUE & 
                           protSmart_raw$ENRICHMENT %in% "CD34" &
                           !is.na(protSmart_raw$Resp_manual)]

protSub$respGroup <- ifelse(protSub$Resp_manual =="R", "R", "NR")
protSub$respGroup <- factor(protSub$respGroup, levels = c("R","NR"))

#only proteins with annotation
protSub <- protSub[!rowData(protSub)$Genes %in% c("",NA),]

#only keep the first symbol if multiple gene names are present
rowData(protSub)$symbol <- sapply(rowData(protSub)$Genes, function(x) {
  str_split(x,";")[[1]][1]
})
